\def \ProveLimGProg {%
    \item Доказать, что при $0 \leq q < 1 \hookrightarrow \lim\limits_{n \to \infty} q^n = 0$.(\textit{ Указание: использовать неравенство Бернулли})
}

\def \ProveLimAbs {%
    \item Доказать, что если $x_n \rightarrow a$, то $|x_n| \rightarrow |a|.$
        
}

\def \ProveLimUniqInf {%
    \item Доказать, что если $\lim\limits_{n \to \infty} x_n = + \infty , $ то у $x_n$ не существует конечного предела.
        
}

\def \ProveWeierstrInf {%
    \item Доказать, что если $x_n$ возрастает и неограничена, то у нее существует предел равный $+ \infty.$
        
}

\def \ProveMonotE {%
    \item Доказать, что $x_n = \Big (1 + \dfrac{1}{n} \Big )^{n + 1}$ монотонна.(\textit{Замечание: в силу теоремы Вейерштрасса это означает, что $x_n$ сходится.})

        
}

\def \ProveLimInequalMinusInf {%
    \item Пусть $ \exists N : \forall n \geq N \hookrightarrow a_n \leq b_n.$ Доказать, что если $b_n \rightarrow - \infty$, то $a_n \rightarrow - \infty$.
        
}

\def \ProveLimInequalPlusInf {%
    \item Пусть $ \exists N : \forall n \geq N \hookrightarrow a_n \leq b_n.$ Доказать, что если $a_n \rightarrow + \infty$, то $b_n \rightarrow + \infty$
        
}

\def \FindLimFactor {%
    \item Найти предел последовательности $x_n = \dfrac{n}{2^n}$ и $y_n = \dfrac{2^n}{n!}$. (\textit{Указание: использовать, что $2^n = (1 + 1)^n = \sum\limits_{k = 0}^n C_n^k $ })
        
}

\def \ExplConv {%
    \item Исследовать на сходимость $x_n = n +(-1)^n$.
        
}

\def \ProveLimSqrt {%
    \item Доказать, что если $x_n \rightarrow a,$ где $a > 0$, то $ \sqrt{x_n} \rightarrow \sqrt{a}.$
        
}

\def \ProveConvE {%
    \item Доказать, что последовательность $x_n = \Big (1 + \dfrac{1}{n} \Big )^n$ сходится.
        
}

\def \FindLimitsEAndSqrt {%
    \item Найти $\lim\limits_{n \to \infty} \Big(9 + \dfrac{1}{n} \Big)^{1/2}$ и $\lim\limits_{n \to \infty} \Big(1 + \dfrac{1}{n} \Big)^{n/2}$.

        
}

\def \ProveAndFindRecur {%
    \item Доказать, что $x_n$ сходится и найти $\lim\limits_{n \to \infty} x_n$, если $x_{n + 1} = x_n^2 + 3x_n + 1$ и $x_1 = -3/2$.(\textit{ Указание: использовать то, что если $x_n \rightarrow a$, то и $x_{n+1} \rightarrow a$})
        
}

\def \FindExample {%
    \item Привести пример когда:\\
        1) $x_n < y_n \forall n \in \mathbb{N},$ но $\lim\limits_{n \to \infty} x_n = \lim\limits_{n \to \infty} y_n.$
        2) $x_n > y_n \forall n \leq 1000,$ но $\lim\limits_{n \to \infty} x_n = \lim\limits_{n \to \infty} y_n.$
        
}
